59template <int numStates, int numInputs>
60void discreteToContinuous(Eigen::Matrix<double, numStates, numStates>& A,
61 Eigen::Matrix<double, numStates, numInputs>& B, double dt) {
62 // Reference paper: https://doi.org/10.1016/0307-904X(80)90177-8
63 auto discA = A;
64 auto discB = B;
65 A = discA.log() / dt;
66 B = A * (discA - Eigen::Matrix<double, numStates, numStates>::Identity()).inverse() *
67 discB;
68}

80template <int numStates, int numInputs>
81void continuousToDiscrete(Eigen::Matrix<double, numStates, numStates>& A,
82 Eigen::Matrix<double, numStates, numInputs>& B, double dt) {
83 // zero order hold discretization for system and input matrices
84 // reference:
85 // https://en.wikipedia.org/wiki/Discretization#Discretization_of_linear_state_space_models
86 Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> M;
87 M.setZero();
88 M.template block<numStates, numStates>(0, 0) = A;
89 M.template block<numStates, numInputs>(0, numStates) = B;
90 M *= dt;
91
92 Eigen::Matrix<double, numStates + numInputs, numStates + numInputs> exp = M.exp();
93 A = exp.template block<numStates, numStates>(0, 0);
94 B = exp.template block<numStates, numInputs>(0, numStates);
95}

124template <int numStates>
125void discretizeAQ(const Eigen::Matrix<double, numStates, numStates>& contA,
126 const Eigen::Matrix<double, numStates, numStates>& contQ,
127 Eigen::Matrix<double, numStates, numStates>& discA,
128 Eigen::Matrix<double, numStates, numStates>& discQ, double dt) {
129 // zero order hold discretization of the system and state covariance matrices
130 // reference:
131 // https://en.wikipedia.org/wiki/Discretization#Discretization_of_process_noise
132 Eigen::Matrix<double, numStates * 2, numStates * 2> M;
133 M.template block<numStates, numStates>(0, 0) = -contA;
134 M.template block<numStates, numStates>(0, numStates) = contQ;
135 M.template block<numStates, numStates>(numStates, 0).setZero();
136 M.template block<numStates, numStates>(numStates, numStates) = contA.transpose();
137 M *= dt;
138
139 Eigen::Matrix<double, numStates * 2, numStates* 2> G = M.exp();
140
141 Eigen::Matrix<double, numStates, numStates> AinvQ =
142 G.block(0, numStates, numStates, numStates); // this is discA^-1 * Q
143 discA = G.block(numStates, numStates, numStates, numStates).transpose();
144
145 Eigen::Matrix<double, numStates, numStates> q = discA * AinvQ; // A * A^-1 * Q = Q
146 discQ = (q + q.transpose()) / 2.0; // make Q symmetric again if it became asymmetric
147}

202template <int numStates>
203static Eigen::Matrix<double, numStates, numStates>
204DARE(const Eigen::Matrix<double, numStates, numStates>& A,
205 const Eigen::Matrix<double, numStates, numStates>& B,
206 const Eigen::Matrix<double, numStates, numStates>& Q,
207 const Eigen::Matrix<double, numStates, numStates>& R, double tolerance = 1e-4,
208 int maxIter = -1) {
209 using mat_t = Eigen::Matrix<double, numStates, numStates>;
210 // see:
211 // https://scicomp.stackexchange.com/questions/30757/discrete-time-algebraic-riccati-equation-dare-solver-in-c
212 // to conform to the wikipedia article conventions, we replace H with P
213
214 mat_t currA = A;
215 mat_t currG = B * R.inverse() * B.transpose();
216 mat_t currP = Q;
217
218 mat_t I = mat_t::Identity();
219
220 double error;
221
222 for (int i = 0; maxIter < 0 || i < maxIter; i++) {
223 mat_t lastA = currA;
224 mat_t lastG = currG;
225 mat_t lastP = currP;
226
227 mat_t AIGHinv = lastA * (I + lastG * lastP).inverse();
228
229 currA = AIGHinv * lastA;
230 currG = lastG + AIGHinv * lastG * lastA.transpose();
231 currP = lastP + lastA.transpose() * lastP * (I + lastG * lastP).inverse() * lastA;
232
233 double error = (currP - lastP).norm() / currP.norm();
234 if (error > tolerance){
235 break;
236 }
237 }
238
239 return currP;
240}
