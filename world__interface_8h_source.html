294template <unsigned long int N>
295types::DataPoint<navtypes::Vectord<N>>
296getMotorPositionsRad(const std::array<types::motorid_t, N>& motors) {
297 navtypes::Vectord<N> motorPositions;
298 std::optional<types::datatime_t> timestamp;
299 for (size_t i = 0; i < motors.size(); i++) {
300 types::DataPoint<int32_t> positionDataPoint = robot::getMotorPos(motors[i]);
301 if (positionDataPoint.isValid()) {
302 if (!timestamp.has_value() || timestamp > positionDataPoint.getTime()) {
303 timestamp = positionDataPoint.getTime();
304 }
305 double position = static_cast<double>(positionDataPoint.getData());
306 position *= M_PI / 180.0 / 1000.0;
307 motorPositions(i) = position;
308 } else {
309 return {};
310 }
311 }
312 return types::DataPoint<navtypes::Vectord<N>>(timestamp.value(), motorPositions);
313}
