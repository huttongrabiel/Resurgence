73 void setTarget(robot::types::datatime_t currTime, const navtypes::Vectord<dim>& startPos,
74 const navtypes::Vectord<dim>& endPos) {
75 profile_t profile{.startTime = currTime, .startPos = startPos, .endPos = endPos};
76 // min time required to accelerate to top speed
77 double rampUpTime = maxVel / maxAccel;
78 // min distance required to accelerate to top speed
79 double rampUpDist = std::pow(maxVel, 2) / (2 * maxAccel);
80
81 double dist = (endPos - startPos).norm();
82 if (2 * rampUpDist < dist) {
83 // the profile distance is too short to allow us to get to top speed
84 // so the profile looks like a triangle instead of a trapezoid
85 profile.stopAccelTime = rampUpTime;
86 double coastTime = (dist - 2 * rampUpDist) / maxVel;
87 profile.startDecelTime = rampUpTime + coastTime;
88 profile.totalTime = coastTime + 2 * rampUpTime;
89 } else {
90 // this is the normal case, where we accelerate to top speed, coast for a bit, then
91 // decelerate.
92 double totalTime = 2 * std::sqrt(dist / maxAccel);
93 profile.stopAccelTime = profile.startDecelTime = totalTime / 2;
94 profile.totalTime = totalTime;
95 }
96 this->profile = profile;
97 }

106 navtypes::Vectord<dim> getCommand(robot::types::datatime_t currTime) const {
107 if (!profile) {
108 return navtypes::Vectord<dim>::Zero();
109 }
110
111 profile_t profile = this->profile.value();
112 navtypes::Vectord<dim> dir = (profile.endPos - profile.startPos).normalized();
113 double elapsedTime = util::durationToSec(currTime - profile.startTime);
114 if (elapsedTime < 0) {
115 return profile.startPos;
116 } else if (elapsedTime < profile.stopAccelTime) {
117 double dist = 0.5 * maxAccel * std::pow(elapsedTime, 2);
118 return profile.startPos + dir * dist;
119 } else if (elapsedTime < profile.startDecelTime) {
120 // area of trapezoid is average of bases times height
121 double dist = (2 * elapsedTime - profile.stopAccelTime) / 2.0 * maxVel;
122 return profile.startPos + dir * dist;
123 } else if (elapsedTime < profile.totalTime) {
124 double distFromEnd = 0.5 * maxAccel * std::pow(profile.totalTime - elapsedTime, 2);
125 return profile.endPos - dir * distFromEnd;
126 } else {
127 return profile.endPos;
128 }
129 }
