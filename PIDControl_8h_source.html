89 // see wikipedia link above for in-depth explanation

91 Arrayd<dim> err = setPoint.value() - currValue;
92 Arrayd<dim> pTerm = err * kP;
93 Arrayd<dim> fTerm = setPoint.value() * kF;
94
95 Arrayd<dim> iTerm = Arrayd<dim>::Zero();
96 Arrayd<dim> dTerm = Arrayd<dim>::Zero();
97
98 if (lastData.has_value()) {
99 // TODO: instead of numerically differentiating we can use a model (if we have one)
100 auto [lastTime, lastValue] = lastData.value();
101 double dtSec = util::durationToSec(currTime - lastTime);
102
103 // use the backwards finite difference for approximating the derivative
104 Arrayd<dim> derivative = (currValue - lastValue) / dtSec;
105 dTerm = derivative * kD;
106
107 // use the trapezoid rule for approximating the integral
108 Arrayd<dim> area = (lastValue + currValue) * dtSec / 2.0;
109 iAccum += area;
110 iTerm = (err.abs() <= iZone).select(iAccum * kI, Arrayd<dim>::Zero());
111 }
112
113 lastData = {currTime, currValue};
114 Arrayd<dim> output = pTerm + iTerm + dTerm + fTerm;
115
116 return output;
117 }
