64 navtypes::Matrixd<2, N> getJacobian(const navtypes::Vectord<N>& jointPos) const {
65 Eigen::Matrix<double, 2, N> jacobian = Eigen::Matrix<double, 2, N>::Zero();
66 std::array<double, N> thetaCumSum;
67 thetaCumSum[0] = jointPos[0];
68 for (unsigned int i = 1; i < N; i++) {
69 thetaCumSum[i] = jointPos[i] + thetaCumSum[i - 1];
70 }
71 for (unsigned int i = 0; i < N; i++) {
72 for (unsigned int j = i; j < N; j++) {
73 jacobian(0, i) += -segLens[j] * std::sin(thetaCumSum[j]);
74 jacobian(1, i) += segLens[j] * std::cos(thetaCumSum[j]);
75 }
76 }
77 return jacobian;
