151template <int stateDim, int inputDim, int processNoiseDim, int numOutputs>
152class MultiSensorEKF : public KalmanFilterBase<stateDim, inputDim> {
153public:
154 using state_t = navtypes::Vectord<stateDim>;
155 using input_t = navtypes::Vectord<inputDim>;
156 using processnoise_t = navtypes::Vectord<processNoiseDim>;
157
158 using statefunc_t =
159 std::function<state_t(const state_t&, const input_t&, const processnoise_t&)>;
160
161 MultiSensorEKF(
162 const statefunc_t& stateFunc,
163 const statespace::NoiseCovMat<stateDim, processNoiseDim, inputDim>& processNoise,
164 double dt, const std::array<Output, numOutputs>& outputs)
165 : stateFunc(stateFunc), Q(processNoise), dt(dt), outputs(outputs) {}
166
167 void predict(const input_t& input) override {
168 navtypes::Matrixd<stateDim, stateDim> F = stateFuncJacobianX(this->xHat, input);
169 navtypes::Matrixd<processNoiseDim, processNoiseDim> processNoise =
170 Q.get(this->xHat, input);
171
172 navtypes::Matrixd<stateDim, processNoiseDim> L =
173 stateFuncJacobianW(this->xHat, input);
174
175 this->xHat = stateFunc(this->xHat, input, processnoise_t::Zero());
176 this->P = F * this->P * F.transpose() + L * processNoise * L.transpose();
177 }
178
179 void correct(int outputIdx, const Eigen::VectorXd& measurement) {
180 Output& output = outputs[outputIdx];
181
182 Eigen::MatrixXd H = output.outputFuncJacobianX(this->xHat);
183 Eigen::MatrixXd outputNoise = output.covMat.get(this->xHat, measurement);
184 Eigen::MatrixXd M = output.outputFuncJacobianV(this->xHat);
185 Eigen::MatrixXd S = H * this->P * H.transpose() +
186 M * outputNoise * M.transpose(); // residual covariance
187 // near-optimal kalman gain
188 Eigen::Matrix<double, stateDim, Eigen::Dynamic> K =
189 S.transpose().colPivHouseholderQr().solve(H * this->P.transpose()).transpose();
190 Eigen::VectorXd y =
191 measurement -
192 output.outputFunc(this->xHat, Eigen::VectorXd::Zero(
193 output.outputNoiseDim)); // measurement residual
194
195 this->xHat = this->xHat + K * y;
196 this->P = (Eigen::Matrix<double, stateDim, stateDim>::Identity() - K * H) * this->P;
197 }
198
199 template <int outputIdx> void correct(const Eigen::VectorXd& measurement) {
200 static_assert(0 <= outputIdx && outputIdx < numOutputs);
201 correct(outputIdx, measurement);
202 }
