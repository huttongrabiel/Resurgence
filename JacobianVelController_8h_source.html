70 navtypes::Vectord<inputDim> getCommand(robot::types::datatime_t currTime,
71 const navtypes::Vectord<inputDim>& currPos,
72 double& cosineSim) {
73 if (!controllerState) {
74 return currPos;
75 }
76
77 // The jacobian of the kinematics function represents a linearization of the kinematics
78 // around the current position. We then use this approximation to find the position
79 // delta in the input space that yields the required position delta in the output space
80 // This is similar to JacobianController, but the command position is calculated
81 // from a velocity instead of a profile.
82
83 navtypes::Vectord<outputDim> currPosOutput = kinematicsFunc(currPos);
84 state_t state = controllerState.value();
85 double dt = util::durationToSec(currTime - state.timestamp);
86 navtypes::Vectord<outputDim> lastTarget = state.lastTarget.value_or(currPosOutput);
87 navtypes::Vectord<outputDim> currTarget = lastTarget + dt * state.targetVel;
88 navtypes::Vectord<outputDim> outputPosDiff = currTarget - currPosOutput;
89 navtypes::Matrixd<outputDim, inputDim> jacobian = jacobianFunc(currPos);
90 navtypes::Vectord<inputDim> inputPosDiff =
91 jacobian.colPivHouseholderQr().solve(outputPosDiff);
92
93 // the cosine similarity of the actual and required position deltas in the output space
94 // is metric of how much the kinematics allows the commanded movement.
95 navtypes::Vectord<outputDim> trueOutputPosDiff = jacobian * inputPosDiff;
96 cosineSim = outputPosDiff.dot(trueOutputPosDiff) /
97 (outputPosDiff.norm() * trueOutputPosDiff.norm());
98
99 state_t newState = {
100 .timestamp = currTime, .targetVel = state.targetVel, .lastTarget = currTarget};
101 controllerState = newState;
102
103 return currPos + inputPosDiff;
104 }
