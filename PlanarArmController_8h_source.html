61 Eigen::Vector2d get_setpoint(robot::types::datatime_t currTime) {
62 Eigen::Vector2d pos;
63 {
64 std::lock_guard<std::mutex> lock(mutex);
65 // calculate current EE setpoint
66 pos = setpoint;
67 if (velTimestamp.has_value()) {
68 double dt = util::durationToSec(currTime - velTimestamp.value());
69 pos += velocity * dt;
70 }
71 }
72
73 // bounds check (new pos + vel vector <= sum of joint lengths)
74 return normalizeEEWithinRadius(pos);
75 }

121 navtypes::Vectord<N> getCommand(robot::types::datatime_t currTime,
122 const navtypes::Vectord<N>& currJointPos) {
123 Eigen::Vector2d newPos = get_setpoint(currTime);
124 // lock after calling get_setpoint since that internally locks the mutex
125 std::lock_guard<std::mutex> lock(mutex);
126 setpoint = newPos;
127
128 // get new joint positions for target EE
129 return kin.eePosToJointPos(newPos, currJointPos);
130 }

146 Eigen::Vector2d normalizeEEWithinRadius(Eigen::Vector2d eePos) {
147 double radius = kin.getSegLens().sum() * safetyFactor;
148
149 if (eePos.norm() > radius) {
150 // TODO: will need to eventually shrink velocity vector until it is within radius
151 // instead of just normalizing it.
152
153 eePos.normalize();
154 eePos *= radius;
155 }
156

158 }
