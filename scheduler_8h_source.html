124 void threadFn() {
125 std::unique_lock lock(scheduleMutex);
126 while (!quitting) {
127 if (schedule.empty()) {
128 scheduleCV.wait(lock, [&] { return newEventAdded; });
129 newEventAdded = false;
130 } else {
131 auto now = Clock::now();
132 schedule_t event = schedule.top();
133 if (toRemove.find(event.id) != toRemove.end()) {
134 // if we should remove this id then remove the event and don't reschedule
135 schedule.pop();
136 toRemove.erase(event.id);
137 } else if (event.nextSendTime <= now) {
138 // execute the event and reschedule it
139 schedule.pop();
140 event.fn();
141 event.nextSendTime += event.period;
142 schedule.push(event);
143 } else {
144 // wait until the event should be executed
145 scheduleCV.wait_until(lock, event.nextSendTime,
146 [&] { return newEventAdded; });
147 newEventAdded = false;
148 }
149 }
150 }
