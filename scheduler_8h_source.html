114 void threadFn() {
115 std::unique_lock lock(scheduleMutex);
116 while (!quitting) {
117 if (schedule.empty()) {
118 scheduleCV.wait(lock, [&] { return newEventAdded; });
119 newEventAdded = false;
120 } else {
121 auto now = Clock::now();
122 schedule_t event = schedule.top();
123 if (toRemove.find(event.id) != toRemove.end()) {
124 // if we should remove this id then remove the event and don't reschedule
125 schedule.pop();
126 toRemove.erase(event.id);
127 } else if (event.nextSendTime <= now) {
128 // execute the event and reschedule it
129 schedule.pop();
130 event.fn();
131 event.nextSendTime += event.period;
132 schedule.push(event);
133 } else {
134 // wait until the event should be executed
135 scheduleCV.wait_until(lock, event.nextSendTime,
136 [&] { return newEventAdded; });
137 newEventAdded = false;
138 }
139 }
140 }
