13template <int D> class Factor : public AbstractFactor {
14public:
15 const covariance<D> _sigma_inv;

17
18 Factor(const covariance<D>& sigma_inv, const measurement<D>& measurement)
19 : _sigma_inv(sigma_inv), _measurement(measurement) {}
20
21 virtual measurement<D> f(const values& /*x*/) = 0;
22 virtual jacobian<D> jf(const values& /*x*/) = 0;
23
24 virtual double eval(const values& x) {
25 measurement<D> diff = f(x) - _measurement;
26 return 0.5 * (diff.transpose() * _sigma_inv * diff)(0, 0);
27 }
28
29 virtual values gradient_at(const values& x) {
30 return jf(x) * (_sigma_inv * (f(x) - _measurement));
31 }
32
33 virtual hessian hessian_at(const values& x) {
34 jacobian<D> j = jf(x);
35 return j * _sigma_inv * j.transpose();
36 }
